//Gerar o arquivo de configuração para o Typescript
npx tsc --init

//Configure o tsconfig.json:
  "rootDir": "./src",
  "outDir": "./dist",

  //Quando essa opção está true, o TypeScript exige que tipos sejam importados separadamente com import type.
  "verbatimModuleSyntax": false,

  //Erros com o Banco para conectar configurando o tsconfig:
  "strictPropertyInitialization": false,
  "experimentalDecorators": true,
  "emitDecoratorMetadata": true,

//Compilar o arquivo Typescript
npx tsc
